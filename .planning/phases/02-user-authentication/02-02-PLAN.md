<!-- API contracts from Plan 01 that this plan calls -->

POST /api/auth (register):
  Request: { email: string, password: string, firstName: string, lastName: string }
  Response 201: { message: "Registration successful", userId: number }
  Response 409: { message: "Email already registered" }
  Response 400: { message: "Email, password, first name, and last name are required" }
  Side effect: Sets HttpOnly cookie "token" (JWT, 7-day expiry)

POST /api/auth/login:
  Request: { email: string, password: string }
  Response 200: { message: "Login successful" }
  Response 401: { message: "Invalid credentials" }
  Response 400: { message: "Email and password are required" }
  Side effect: Sets HttpOnly cookie "token" (JWT, 7-day expiry)

GET /api/auth/me (requires cookie):
  Response 200: { id: number, email: string, firstName: string, lastName: string }
  Response 401: { message: "Unauthorized" }
