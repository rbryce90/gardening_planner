       - Create module-level: const secret = new TextEncoder().encode(Deno.env.get("JWT_SECRET")!)
       - export async function signToken(userId: number, email: string): Promise<string>
         - return await new SignJWT({ userId, email }).setProtectedHeader({ alg: "HS256" }).setIssuedAt().setExpirationTime("7d").sign(secret)
       - export async function verifyToken(token: string): Promise<{ userId: number; email: string }>
         - const { payload } = await jwtVerify(token, secret)
         - return payload as { userId: number; email: string }
       - export async function register(email: string, password: string, firstName: string, lastName: string): Promise<{ userId: number; token: string }>
         - Calls userRepository.createUser(email, password, firstName, lastName) to get userId
         - Calls signToken(userId, email) to get token
         - Returns { userId, token }
         - Let any UNIQUE constraint error from SQLite propagate (caught by route as 409)
       - export async function login(email: string, password: string): Promise<{ userId: number; token: string }>
         - Calls userRepository.findByEmail(email)
         - If no user found: throw Object.assign(new Error("Invalid credentials"), { status: 401 })
         - Calls comparePassword(password, user.password)
         - If not valid: throw Object.assign(new Error("Invalid credentials"), { status: 401 })
         - Calls signToken(user.id, email) to get token
         - Returns { userId: user.id, token }

    2. Create server/middleware/auth.ts:
       - Import { Request, Response, NextFunction } from "express"
